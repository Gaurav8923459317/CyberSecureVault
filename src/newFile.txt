import React, { useState } from "react";
import "./FileUpload.css";

/**
 * FileUpload Component - File Upload Interface
 * Features:
 * - Drag & Drop file upload
 * - File preview (images, text, PDF)
 * - Auto redirect to dashboard after upload
 * - Save uploaded files metadata to localStorage
 */
function FileUpload({ onBack }) {
  // ========== STATE DEFINITIONS ==========
  const [selectedFiles, setSelectedFiles] = useState([]);        // User selected files
  const [uploadProgress, setUploadProgress] = useState(0);       // Upload progress percentage
  const [isDragOver, setIsDragOver] = useState(false);           // Drag & drop highlight state
  const [previewFile, setPreviewFile] = useState(null);          // File for preview modal
  const [textContent, setTextContent] = useState("");            // Text file content for preview

  // Ref to reset file input after clear
  const fileInputRef = React.useRef(null);

  // ========== FILE PREVIEW FUNCTIONS ==========
  
  /**
   * Handle file preview when user clicks on file card
   * Supports: Images, Text files, PDF, Other files
   */
  const handleFilePreview = (file) => {
    setPreviewFile(file);

    // Text files ke liye content read karen
    if (file.type.startsWith("text/") || file.name.endsWith(".txt")) {
      const reader = new FileReader();
      reader.onload = (e) => {
        setTextContent(e.target.result);
      };
      reader.readAsText(file);
    } else {
      setTextContent("");
    }
  };

  /**
   * Close file preview modal
   */
  const handleClosePreview = () => {
    setPreviewFile(null);
  };

  // ========== FILE SELECTION FUNCTIONS ==========
  
  /**
   * Handle file selection from input field
   */
  const handleFileSelect = (event) => {
    const files = Array.from(event.target.files);
    setSelectedFiles(files);
    setUploadProgress(0); // Reset progress on new selection
  };

  /**
   * Handle drag over event for drag & drop
   */
  const handleDragOver = (event) => {
    event.preventDefault();
    setIsDragOver(true);
  };

  /**
   * Handle drag leave event for drag & drop
   */
  const handleDragLeave = (event) => {
    event.preventDefault();
    setIsDragOver(false);
  };

  /**
   * Handle file drop for drag & drop
   */
  const handleDrop = (event) => {
    event.preventDefault();
    setIsDragOver(false);
    const files = Array.from(event.dataTransfer.files);
    setSelectedFiles(files);
    setUploadProgress(0);
  };

  // ========== UPLOAD FUNCTIONS ==========
  
  /**
   * Simulate file upload with auto redirect and file saving
   * Steps:
   * 1. Show upload progress
   * 2. Save file metadata to localStorage (no duplicates)
   * 3. Show success message
   * 4. Auto redirect to dashboard after 2 seconds
   */
 const handleUpload = () => {
  if (selectedFiles.length === 0) return;

  // Simulate upload progress (0% to 100%)
  let progress = 0;
  const interval = setInterval(() => {
    progress += 10;
    setUploadProgress(progress);
    
    if (progress >= 100) {
      clearInterval(interval);
      
      // ✅ STEP 1: Uploaded files ko localStorage mein save karen
      const uploadedFilesInfo = selectedFiles.map(file => ({
        name: file.name,
        size: file.size,
        type: file.type,
        lastModified: file.lastModified,
        id: `${file.name}_${file.lastModified}_${file.size}` // Unique ID
      }));
      
      // ✅ STEP 2: Check karen konsi files already uploaded hain
      const existingFiles = JSON.parse(localStorage.getItem('uploadedFiles') || '[]');
      const existingFileIds = new Set(existingFiles.map(f => f.id));
      
      // ✅ STEP 3: Separate new files and duplicate files
      const newFiles = [];
      const duplicateFiles = [];
      
      uploadedFilesInfo.forEach(file => {
        if (existingFileIds.has(file.id)) {
          duplicateFiles.push(file.name); // Duplicate file names store karen
        } else {
          newFiles.push(file); // New files store karen
        }
      });
      
      // ✅ STEP 4: New files ko existing files ke saath merge karen
      if (newFiles.length > 0) {
        const allFiles = [...existingFiles, ...newFiles];
        localStorage.setItem('uploadedFiles', JSON.stringify(allFiles));
      }
      
      // ✅ STEP 5: SMART MESSAGE BASED ON FILE STATUS
      let message = "";
      
      if (newFiles.length > 0 && duplicateFiles.length > 0) {
        // Some new, some duplicate
        message = `✅ Upload Complete!\n\n📁 New Files Uploaded: ${newFiles.length}\n- ${newFiles.map(f => f.name).join('\n- ')}\n\n⚠️ Already Uploaded (Skipped): ${duplicateFiles.length}\n- ${duplicateFiles.join('\n- ')}`;
      } 
      else if (newFiles.length > 0) {
        // All new files
        message = `✅ Files Uploaded Successfully!\n\n📁 New Files: ${newFiles.length}\n- ${newFiles.map(f => f.name).join('\n- ')}`;
      }
      else if (duplicateFiles.length > 0) {
        // All duplicate files
        message = `⚠️ Files Already Uploaded!\n\n📁 These files are already in your uploaded list:\n- ${duplicateFiles.join('\n- ')}\n\nNo new files were added.`;
      }
      
      // ✅ STEP 6: Show appropriate message
      alert(message + `\n\nRedirecting to dashboard in 2 seconds...`);
      
      // Auto redirect after 2 seconds
      setTimeout(() => {
        onBack(); // Dashboard par redirect
      }, 2000);
    }
  }, 200);
};

  /**
   * Clear selected files and reset form
   */
  const handleClear = () => {
    setSelectedFiles([]);
    setUploadProgress(0);
    // File input reset karenge
    if (fileInputRef.current) {
      fileInputRef.current.value = '';
    }
  };

  // ========== COMPONENT RENDER ==========
  return (
    <div className="file-upload">
      
      {/* ========== HEADER SECTION ========== */}
      <div className="upload__header">
        <button className="back__btn" onClick={onBack}>
          ← Back to Dashboard
        </button>
        <h1>📤 Upload Files</h1>
        <p>Select files to encrypt and secure</p>
      </div>

      {/* ========== DRAG & DROP AREA ========== */}
      <div
        className={`upload__area ${isDragOver ? "upload__area--dragover" : ""}`}
        onDragOver={handleDragOver}
        onDragLeave={handleDragLeave}
        onDrop={handleDrop}
      >
        <div className="upload__content">
          <div className="upload__icon">📁</div>
          <h3>Drag & Drop Files Here</h3>
          <p>or</p>

          {/* File Input - Hidden but triggered by label */}
          <label htmlFor="file-input" className="upload__browse-btn">
            Browse Files
          </label>
          <input
            id="file-input"
            type="file"
            multiple
            onChange={handleFileSelect}
            className="upload__input"
            ref={fileInputRef}
          />
        </div>
      </div>

      {/* ========== SELECTED FILES LIST ========== */}
      {selectedFiles.length > 0 && (
        <div className="files__list">
          <h3>Selected Files ({selectedFiles.length})</h3>

          {/* Files Grid - Shows all selected files */}
          <div className="files__grid">
            {selectedFiles.map((file, index) => (
              <div
                key={index}
                className="file__card"
                onClick={() => handleFilePreview(file)}
              >
                {/* File type based icons */}
                <div className="file__icon">
                  {file.type.startsWith("image/")
                    ? "🖼️"
                    : file.type === "application/pdf"
                    ? "📄"
                    : file.type.startsWith("text/")
                    ? "📝"
                    : "📁"}
                </div>
                <div className="file__info">
                  <h4>{file.name}</h4>
                  <p>{(file.size / 1024 / 1024).toFixed(2)} MB</p>
                  <span className="file__preview-hint">
                    Click to preview
                  </span>
                </div>
              </div>
            ))}
          </div>

          {/* ========== UPLOAD PROGRESS ========== */}
          {uploadProgress > 0 && (
            <div className="upload__progress">
              <div className="progress__bar">
                <div
                  className="progress__fill"
                  style={{ width: `${uploadProgress}%` }}
                ></div>
              </div>
              <span>{uploadProgress}%</span>
            </div>
          )}

          {/* ========== ACTION BUTTONS ========== */}
          <div className="upload__actions">
            <button 
              className="btn btn--primary" 
              onClick={handleUpload}
              disabled={uploadProgress > 0 && uploadProgress < 100}
            >
              {uploadProgress === 0 ? 'Upload Files' : 'Uploading...'}
            </button>

            <button className="btn btn--secondary" onClick={handleClear}>
              Clear Files
            </button>
          </div>
        </div>
      )}

      {/* ========== FILE PREVIEW MODAL ========== */}
      {previewFile && (
        <div className="preview__overlay">
          <div className="preview__modal">
            <div className="preview__header">
              <h3>File Preview: {previewFile.name}</h3>
              <button className="preview__close" onClick={handleClosePreview}>
                ✕
              </button>
            </div>
            <div className="preview__content">
              
              {/* Image File Preview */}
              {previewFile.type.startsWith("image/") && (
                <img
                  src={URL.createObjectURL(previewFile)}
                  alt="Preview"
                  className="preview__image"
                />
              )}

              {/* Text File Preview */}
              {(previewFile.type.startsWith("text/") ||
                previewFile.name.endsWith(".txt")) && (
                <div className="preview__text">
                  <h4>Text Content:</h4>
                  <textarea
                    className="preview__textarea"
                    value={textContent}
                    readOnly
                    rows={Math.min(25, textContent.split("\n").length + 5)}
                    style={{
                      height: "auto",
                      minHeight: "200px",
                      maxHeight: "60vh",
                    }}
                  />
                </div>
              )}

              {/* PDF File Preview */}
              {previewFile.type === "application/pdf" && (
                <div className="preview__info">
                  <p>📄 PDF File</p>
                  <p><strong>Name:</strong> {previewFile.name}</p>
                  <p><strong>Size:</strong> {(previewFile.size / 1024 / 1024).toFixed(2)} MB</p>
                  <p>PDF preview requires PDF.js library</p>
                </div>
              )}

              {/* Other File Types Preview */}
              {!previewFile.type.startsWith("image/") &&
                !previewFile.type.startsWith("text/") &&
                previewFile.type !== "application/pdf" && (
                  <div className="preview__info">
                    <p>📁 File Information</p>
                    <p><strong>Name:</strong> {previewFile.name}</p>
                    <p><strong>Type:</strong> {previewFile.type || "Unknown"}</p>
                    <p><strong>Size:</strong> {(previewFile.size / 1024 / 1024).toFixed(2)} MB</p>
                    <p>Download option can be added here</p>
                  </div>
                )}
            </div>
          </div>
        </div>
      )}
    </div>
  );
}

export default FileUpload;

   * Load uploaded files from localStorage when component mounts

        setPassword('');
        setEncryptionProgress(0);
      }
    }, 200);
